# pitch_service.py
import librosa
import numpy as np
from fastapi import FastAPI

app = FastAPI()

class PitchAnalyzer:
def __init__(self):
self.sr = 22050  # 采样率

    async def analyze(self, audio_path: str) -> dict:
        # 加载音频
        y, sr = librosa.load(audio_path, sr=self.sr)
        
        # 方法1：使用 PYIN 算法（精度高）
        f0, voiced_flag, voiced_probs = librosa.pyin(
            y, 
            fmin=80,      # 最低音 C2
            fmax=500,     # 最高音 B4
            sr=sr
        )
        
        # 过滤静音
        pitches = f0[voiced_flag]
        if len(pitches) == 0:
            return self._fallback_detection(y, sr)
        
        # 转换为 MIDI 编号
        midi_pitches = 69 + 12 * np.log2(pitches / 440)
        
        # 计算音域
        low_note = int(np.min(midi_pitches))
        high_note = int(np.max(midi_pitches))
        
        # 计算平均音高
        mean_pitch = float(np.mean(midi_pitches))
        
        # 音高变化幅度（标准差）
        pitch_std = float(np.std(midi_pitches))
        
        # 音高变化率（一阶差分）
        pitch_diff = np.diff(midi_pitches)
        high_jump = np.max(np.abs(pitch_diff))
        
        return {
            "lowest_note": low_note,
            "lowest_note_name": self._midi_to_note(low_note),
            "highest_note": high_note,
            "highest_note_name": self._midi_to_note(high_note),
            "vocal_range": high_note - low_note,
            "mean_pitch": mean_pitch,
            "pitch_variation": pitch_std,
            "max_jump": high_jump,
            "pitch_sequence": midi_pitches.tolist()[:100]  # 采样前100个点
        }

# rhythm_service.py
import librosa
import numpy as np

class RhythmAnalyzer:
async def analyze(self, audio_path: str) -> dict:
y, sr = librosa.load(audio_path)

        # 检测 BPM
        tempo, beats = librosa.beat.beat_track(y=y, sr=sr)
        
        # 检测节拍点
        beat_frames = librosa.beat.beat_track(y=y, sr=sr, units='frames')[1]
        beat_times = librosa.frames_to_time(beat_frames, sr=sr)
        
        # 节奏复杂度（非整数倍节拍比例）
        onset_env = librosa.onset.onset_strength(y=y, sr=sr)
        onset_frames = librosa.onset.onset_detect(onset_envelope=onset_env, sr=sr)
        syncopation = len(onset_frames) / len(beat_frames)
        
        # 节奏型分类
        if tempo < 80:
            rhythm_type = "slow_ballad"
        elif tempo < 120:
            rhythm_type = "medium"
        else:
            rhythm_type = "fast"
        
        return {
            "bpm": float(tempo),
            "beat_count": len(beat_times),
            "duration_seconds": len(y) / sr,
            "syncopation_score": float(syncopation),  # 切分音密集度
            "rhythm_type": rhythm_type,
            "difficulty_factor": self._calc_rhythm_difficulty(tempo, syncopation)
        }
    
    def _calc_rhythm_difficulty(self, tempo, syncopation):
        """节奏难度评分（1-10）"""
        tempo_score = min(5, tempo / 30)
        sync_score = min(5, syncopation * 5)
        return min(10, tempo_score + sync_score)

# structure_service.py
import librosa
import numpy as np
from sklearn.cluster import KMeans

class StructureAnalyzer:
async def analyze(self, audio_path: str) -> dict:
y, sr = librosa.load(audio_path)

        # 提取 MFCC 特征
        mfcc = librosa.feature.mfcc(y=y, sr=sr, n_mfcc=13)
        
        # 自相似矩阵（SSM）
        similarity = librosa.segment.recurrence_matrix(mfcc)
        
        # 检测边界
        boundary_frames = librosa.segment.agglomerative(
            mfcc.T, 16, power=2
        )
        boundary_times = librosa.frames_to_time(boundary_frames, sr=sr)
        
        # 段落分类（简化版）
        segments = self._classify_segments(y, sr, boundary_times)
        
        return {
            "segments": segments,
            "segment_count": len(segments),
            "intro_duration": segments[0].get('duration', 0) if segments else 0,
            "outro_duration": segments[-1].get('duration', 0) if segments else 0
        }
    
    def _classify_segments(self, y, sr, boundaries):
        """对段落进行分类（简单启发式）"""
        segments = []
        for i, boundary in enumerate(boundaries):
            start = boundaries[i-1] if i > 0 else 0
            end = boundary
            
            # 计算该段落的能量和频谱特征
            segment = y[int(start*sr):int(end*sr)]
            energy = np.mean(segment**2)
            
            # 简单分类
            if energy < 0.01:
                seg_type = "intro/outro"
            elif i == len(boundaries)//2:
                seg_type = "chorus"
            else:
                seg_type = "verse"
            
            segments.append({
                "type": seg_type,
                "start": round(start, 1),
                "end": round(end, 1),
                "duration": round(end - start, 1),
                "energy": round(float(energy), 4)
            })
        return segments
3.4 难度评分服务 (Difficulty Scorer)
职责：综合所有维度，给出 1-10 级难度评分

# difficulty_service.py

class DifficultyScorer:
def __init__(self):
# 权重配置
self.weights = {
'vocal_range': 0.30,      # 音域宽度
'highest_note': 0.20,     # 最高音
'bpm': 0.15,              # 速度
'pitch_variation': 0.10,  # 音高变化
'syncopation': 0.10,      # 切分音
'duration': 0.05,         # 歌曲长度
'breath_requirement': 0.10 # 气息要求
}

    async def score(self, analysis_result: dict) -> dict:
        """综合评分"""
        
        # 音域评分（音域越宽分越高）
        vocal_range = analysis_result['pitch']['vocal_range']
        range_score = min(10, vocal_range / 2)
        
        # 最高音评分（C5以上开始加分）
        highest = analysis_result['pitch']['highest_note']
        high_score = max(0, (highest - 72) / 3) if highest > 72 else 0
        
        # 速度评分
        bpm = analysis_result['rhythm']['bpm']
        tempo_score = min(8, bpm / 15)
        
        # 综合难度
        total_score = (
            range_score * self.weights['vocal_range'] +
            high_score * self.weights['highest_note'] +
            tempo_score * self.weights['bpm']
        )
        
        # 归一化到 1-10
        final_score = min(10, max(1, round(total_score)))
        
        # 难度等级
        if final_score <= 3:
            level = "beginner"
        elif final_score <= 6:
            level = "intermediate"
        else:
            level = "advanced"
        
        return {
            "total": final_score,
            "level": level,
            "level_name": self._get_level_name(level),
            "breakdown": {
                "vocal_range": round(range_score, 1),
                "highest_note": round(high_score, 1),
                "tempo": round(tempo_score, 1)
            }
        }
    
    def _get_level_name(self, level):
        return {
            "beginner": "初学者",
            "intermediate": "进阶",
            "advanced": "专业级"
        }.get(level, level)

python
# main.py
from fastapi import FastAPI, File, UploadFile, BackgroundTasks
from fastapi.responses import JSONResponse
import uuid
import asyncio

app = FastAPI(title="WanaSing Song Analyzer")

# 初始化各服务
pitch_analyzer = PitchAnalyzer()
rhythm_analyzer = RhythmAnalyzer()
structure_analyzer = StructureAnalyzer()
difficulty_scorer = DifficultyScorer()

@app.post("/api/v1/analyze")
async def analyze_song(
file: UploadFile = File(...),
background_tasks: BackgroundTasks = None
):
"""上传歌曲进行分析"""

    # 生成任务ID
    task_id = str(uuid.uuid4())
    
    # 保存临时文件
    temp_path = f"/tmp/{task_id}.mp3"
    content = await file.read()
    with open(temp_path, "wb") as f:
        f.write(content)
    
    # 并行执行三个分析任务
    pitch_task = asyncio.create_task(pitch_analyzer.analyze(temp_path))
    rhythm_task = asyncio.create_task(rhythm_analyzer.analyze(temp_path))
    structure_task = asyncio.create_task(structure_analyzer.analyze(temp_path))
    
    # 等待所有任务完成
    pitch_result, rhythm_result, structure_result = await asyncio.gather(
        pitch_task, rhythm_task, structure_task
    )
    
    # 综合评分
    combined_result = {
        "pitch": pitch_result,
        "rhythm": rhythm_result,
        "structure": structure_result
    }
    difficulty_result = await difficulty_scorer.score(combined_result)
    
    # 生成最终报告
    report = {
        "task_id": task_id,
        "song_info": {
            "filename": file.filename,
            "duration": rhythm_result.get("duration_seconds", 0)
        },
        "analysis": combined_result,
        "difficulty": difficulty_result,
        "tips": self._generate_tips(combined_result, difficulty_result)
    }
    
    # 清理临时文件
    background_tasks.add_task(self._cleanup, temp_path)
    
    return JSONResponse(content=report)

def _generate_tips(self, analysis, difficulty):
"""生成演唱技巧提示"""
tips = []

    # 音域相关提示
    if analysis['pitch']['vocal_range'] > 12:
        tips.append("这首歌音域较宽，注意换声区的平滑过渡")
    
    # 高音提示
    if analysis['pitch']['highest_note'] > 72:  # C5
        tips.append("副歌有较高音区，建议用混声/假声演唱")
    
    # 节奏提示
    if analysis['rhythm']['syncopation_score'] > 2:
        tips.append("切分音较多，注意节奏的准确度")
    
    return tips[:3]  # 最多返回3条
